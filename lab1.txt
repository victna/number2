#include <stdio.h>
void getArrayInfo(int *arr,int n, int *uniqueCount, int *min, int *max);
int main(void){
    int arr[] = {3,1,4,1,5,9,2,6,5,3,5};
    int n = sizeof(arr)/sizeof(arr[0]);
    int uniqueCount, min, max;
    getArrayInfo(arr, n, &uniqueCount, &min, &max);
    printf("Unique Count: %d\n", uniqueCount);
    printf("Min: %d\n", min);
    printf("Max: %d\n", max);
    return 0;
}
void getArrayInfo(int *arr,int n, int *uniqueCount, int *min, int *max){ 
    if(n<=0){
        *uniqueCount=0;
        return;
    }
    int count = 1, tempMin = arr[0], tempMax = arr[0];
    for(int i=1;i<n;i++){
        if(arr[i]<tempMin){
            tempMin=arr[i];
        }else{
            if(arr[i]>tempMax){
                tempMax=arr[i];
            }
        }
        int r;
            for(r=0;r<i;r++){
                if(arr[i]==arr[r]) break;
            }
            if(r==i) count++;
        }
    *uniqueCount = count;
    *min = tempMin;
    *max = tempMax;
}
